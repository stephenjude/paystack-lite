<script src="https://js.paystack.co/v1/inline.js"></script>
<script>
function payWithPaystack(amount, email, meta, callback, onclose) {
   var meta_data = meta ? meta : {};
    var options = {
        key: "",
        email: email,
        amount: amount+'00',
        metadata: meta,
        callback: function(response){
            callback(response);
        },
        onClose:function(){
            if(onclose){
                onclose();
            }
        }
    };

    var handler = PaystackPop.setup(options);
    handler.openIframe();
}
</script>
